    import { Box as InkBox, type BoxProps as InkBoxProps } from 'ink';

    export interface SunBoxProps extends InkBoxProps {
      padding?: keyof typeof theme.spacing;
      gap?: keyof typeof theme.spacing;
    }

    export const SunBox: React.FC<SunBoxProps> = ({ padding, gap, ...props }) => {
      const resolvedPadding = padding ? theme.spacing[padding] : undefined;
      const resolvedGap = gap ? theme.spacing[gap] : undefined;

    };

    **packages/core/src/ui/adapters/SilentUiAdapter.ts** (per D-39):
    ```typescript
    import type { UiAdapter, UiPattern, UiOutcome, UiPatch } from './UiAdapter.js';

    export class SilentUiAdapter implements UiAdapter {
      async mountPattern(pattern: UiPattern): Promise<UiOutcome> {
        if (pattern.kind === 'choice') {
          const data = pattern.data as { options: Array<{id: string; label: string}>; defaultId?: string };
          const selected = data.defaultId
            ? data.options.find(o => o.id === data.defaultId) ?? data.options[0]
            : data.options[0];
          return {
            kind: 'choice',
            result: { selectedId: selected.id, selectedLabel: selected.label, source: 'noninteractive' as const },
          };
        }
        return { kind: pattern.kind, result: undefined };
      }

      update(_handleId: string, _patch: UiPatch): void { /* no-op */ }
      dispose(_handleId: string): void { /* no-op */ }
    }
    ```
