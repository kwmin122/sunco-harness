**3. Crash recovery** — at start of execute(), before phase loop:
```typescript
// Initialize operational resilience
const sunDir = join(ctx.cwd, '.sun');
const lock = new AutoLock(sunDir);
const stuckDetector = new StuckDetector();

// Read budget ceiling from config
const budgetCeiling = (ctx.config?.get?.('budget_ceiling') as number | undefined) ?? null;
const budgetGuard = new BudgetGuard(budgetCeiling);

// Read timeout config
const supervisorConfig = (ctx.config?.get?.('auto_supervisor') as Record<string, number> | undefined) ?? {};
const hardTimeoutMs = (supervisorConfig.hard_timeout_minutes ?? 30) * 60 * 1000;
const idleTimeoutMs = (supervisorConfig.idle_timeout_minutes ?? 10) * 60 * 1000;

// Check for crashed previous session
const lockCheck = await lock.check();
if (lockCheck.crashed && lockCheck.state) {
  ctx.log.warn('Crashed session detected', { state: lockCheck.state });
  // Find the phase and step where it crashed
  startPhase = lockCheck.state.phase;
  // Skip to the step that was interrupted
  ctx.log.info(`Resuming from Phase ${lockCheck.state.phase}, step: ${lockCheck.state.step}`);
}

// Acquire lock for this session
await lock.acquire(startPhase, 'init');
```

**4. In the step loop** — wrap ctx.run() with timeout + tracking:
```typescript
// Before each step
await lock.updateStep(phaseNumber, step.name);

// Budget check before agent dispatch
const tracker = ctx.agent.getTracker?.();
if (tracker) {
  const budgetResult = budgetGuard.check(tracker.getTotalCostUsd());
  if (budgetResult.status === 'exceeded') {
    ctx.log.error('Budget ceiling reached', budgetResult);
    aborted = true;
    break;
  }
  if (budgetResult.message) {
    ctx.log.warn(budgetResult.message);
  }
}

// Run with hard timeout
const result = await Promise.race([
  ctx.run(step.skillId, { phase: phaseNumber }),
  new Promise<SkillResult>((_, reject) =>
    setTimeout(() => reject(new Error(`Hard timeout: ${step.name} exceeded ${hardTimeoutMs / 60000} minutes`)), hardTimeoutMs)
  ),
]);

// Record invocation for stuck detection
await lock.recordInvocation(step.skillId, result.success);

// Check for stuck state
const history = await lock.getHistory();
const stuckResult = stuckDetector.analyze(history);
if (stuckResult.stuck) {
  ctx.log.error('Stuck detected', stuckResult);
  // Try debug diagnostic
  try {
    await ctx.run('workflow.debug', { phase: phaseNumber });
  } catch { /* debug failed too */ }
  aborted = true;
  break;
}
```

```typescript
// Cost tracking from usage history in state
const usageHistory = await ctx.state.get('usage.history') as UsageEntry[] | null;
if (usageHistory && usageHistory.length > 0) {
  // Per-skill breakdown
  const bySkill = new Map<string, number>();
  let totalCost = 0;
  for (const entry of usageHistory) {
    bySkill.set(entry.skillId, (bySkill.get(entry.skillId) ?? 0) + entry.costUsd);
    totalCost += entry.costUsd;
  }

  details.push('', 'Cost breakdown:');
  for (const [skill, cost] of [...bySkill.entries()].sort((a, b) => b[1] - a[1])) {
    details.push(`  ${skill}: $${cost.toFixed(4)}`);
  }
  details.push(`  Total: $${totalCost.toFixed(4)}`);
}
```
