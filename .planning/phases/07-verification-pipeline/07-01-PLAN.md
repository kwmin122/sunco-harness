```typescript
// VerifyFinding: individual finding from any layer/source
export interface VerifyFinding {
  layer: number;
  source: 'security' | 'performance' | 'architecture' | 'correctness' | 'lint' | 'guard' | 'tribal' | 'acceptance' | 'scope' | 'adversarial' | 'intent' | 'scenario';
  severity: 'critical' | 'high' | 'medium' | 'low';
  description: string;
  file?: string;
  line?: number;
  suggestion?: string;
  humanRequired?: boolean; // REV-03: tribal/regulatory flags
}

export type VerifyVerdict = 'PASS' | 'WARN' | 'FAIL';

export interface LayerResult {
  layer: number;
  name: string;
  findings: VerifyFinding[];
  passed: boolean;
  durationMs: number;
}

export interface VerifyReport {
  verdict: VerifyVerdict;
  layers: LayerResult[];
  findings: VerifyFinding[];
  humanGateRequired: boolean;
  timestamp: string;
}

// Coverage types for validate skill (D-13, D-14, D-15)
export interface CoverageMetric {
  total: number;
  covered: number;
  skipped: number;
  pct: number;
}

export interface FileCoverage {
  path: string;
  lines: CoverageMetric;
  statements: CoverageMetric;
  branches: CoverageMetric;
  functions: CoverageMetric;
}

export interface CoverageReport {
  overall: {
    lines: CoverageMetric;
    statements: CoverageMetric;
    branches: CoverageMetric;
    functions: CoverageMetric;
  };
  files: FileCoverage[];
  uncoveredFiles: string[];
  previousSnapshot?: CoverageReport['overall'];
  delta?: {
    lines: number;
    statements: number;
    branches: number;
    functions: number;
  };
}
```

- Export `parseCoverageSummary(jsonContent: string, previousSnapshot?: CoverageReport['overall']): CoverageReport`
- Parse Istanbul-format coverage-summary.json (keys are file paths + "total")
- Extract overall from data.total
- Build files array from all non-"total" keys, each with lines/statements/branches/functions CoverageMetric
- Identify uncoveredFiles where lines.pct === 0
- If previousSnapshot provided, compute delta for each metric (current.pct - previous.pct)
- Handle edge case: empty JSON with only "total" key returns empty files array

**Per D-10, D-12 from CONTEXT.md**, the 4 expert prompts:

1. `verify-security.ts` -- Export `buildVerifySecurityPrompt(diff: string): string`
   Focus areas: injection (SQL, command, XSS, template), authentication/authorization (missing auth checks, privilege escalation, token handling), data exposure (secrets in code, PII logging, error leaks), input validation, dependency risks.
   JSON output: `{ "findings": [{ "severity", "description", "file?", "line?", "suggestion?" }] }`

2. `verify-performance.ts` -- Export `buildVerifyPerformancePrompt(diff: string): string`
   Focus areas: algorithmic complexity (O(n^2), O(n^3)), memory leaks (unclosed handles, growing caches), N+1 query patterns, unnecessary allocations, blocking I/O in hot paths, missing pagination.
   Same JSON output format.

3. `verify-architecture.ts` -- Export `buildVerifyArchitecturePrompt(diff: string): string`
   Focus areas: coupling violations (circular deps, God objects), layer breaches (UI importing DB), abstraction leaks (implementation details exposed), naming inconsistencies, missing interfaces.
   Same JSON output format.

4. `verify-correctness.ts` -- Export `buildVerifyCorrectnessPrompt(diff: string): string`
   Focus areas: logic errors (off-by-one, wrong comparator), edge cases (null, empty, boundary), data flow issues (unhandled promises, race conditions), type narrowing gaps, incomplete error handling.
   Same JSON output format.

6. `verify-adversarial.ts` -- Export `buildVerifyAdversarialPrompt(diff: string, intentContext: string): string`
   Receives diff + original intent (from CONTEXT.md). Role: "devil's advocate" -- find ways the implementation fails to meet intent. Not just code quality but intent alignment.
   JSON output: `{ "findings": [{ "severity", "description", "intentMismatch": "string", "file?", "suggestion?" }] }`
