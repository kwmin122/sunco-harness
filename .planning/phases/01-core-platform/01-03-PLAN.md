export interface StateApi {
  get<T = unknown>(key: string): Promise<T | null>;
  set(key: string, value: unknown): Promise<void>;
  delete(key: string): Promise<void>;
  list(prefix?: string): Promise<string[]>;
}

export interface FileStoreApi {
  read(relativePath: string): Promise<string | null>;
  write(relativePath: string, content: string): Promise<void>;
  exists(relativePath: string): Promise<boolean>;
  list(directory: string): Promise<string[]>;
}

export interface StateEngine {
  db: StateApi;
  files: FileStoreApi;
  initialize(sunDir: string): Promise<void>;
  close(): void;
}

  <behavior>
    **directory.ts:**
    - initSunDirectory(basePath) creates .sun/ with subdirs: state/, rules/, tribal/, scenarios/, planning/
    - initSunDirectory(basePath) creates .sun/.gitignore containing "state.db\nstate.db-wal\nstate.db-shm\n"
    - initSunDirectory is idempotent (calling twice does not error)
    - ensureSunDir(basePath) returns full path to .sun/ dir, creating if needed

    **database.ts:**
    - createDatabase(dbPath) returns a StateDatabase instance
    - Database opens in WAL mode: PRAGMA journal_mode=WAL
    - Database sets busy_timeout=5000: PRAGMA busy_timeout=5000
    - Database sets synchronous=NORMAL: PRAGMA synchronous=NORMAL
    - Database creates a `state` table: CREATE TABLE IF NOT EXISTS state (key TEXT PRIMARY KEY, value TEXT NOT NULL, updated_at TEXT NOT NULL)
    - db.get("nonexistent") => null
    - db.set("key1", {a: 1}) then db.get("key1") => {a: 1} (JSON serialization)
    - db.set("key1", "v1") then db.set("key1", "v2") then db.get("key1") => "v2" (upsert)
    - db.delete("key1") then db.get("key1") => null
    - db.list() => all keys
    - db.list("agent.") => only keys starting with "agent."
    - db.close() closes the database connection

    **file-store.ts:**
    - FileStore constructor takes sunDir path
    - store.read("rules/arch.md") reads .sun/rules/arch.md, returns content or null if missing
    - store.write("rules/arch.md", content) writes to .sun/rules/arch.md, creating parent dirs
    - store.exists("rules/arch.md") returns true/false
    - store.list("rules") returns filenames in .sun/rules/
    - Path traversal protection: store.read("../../../etc/passwd") throws error

    **api.ts:**
    - createStateEngine(sunDir) returns StateEngine with db + files
    - initialize(sunDir) calls initSunDirectory + createDatabase
    - close() closes database
  </behavior>
  <implementation>
    **directory.ts:**
    - import fs/promises, path
    - initSunDirectory: mkdir -p for each SUN_DIR_STRUCTURE value
    - Write .gitignore to .sun/ for db files

    **database.ts:**
    - import Database from 'better-sqlite3'
    - createDatabase(dbPath): new Database(dbPath), execute PRAGMAs, CREATE TABLE
    - StateDatabase class wrapping better-sqlite3 Database:
      - get(key): SELECT value FROM state WHERE key = ?; parse JSON
      - set(key, value): INSERT OR REPLACE INTO state (key, value, updated_at) VALUES (?, ?, datetime('now'))
      - delete(key): DELETE FROM state WHERE key = ?
      - list(prefix?): SELECT key FROM state WHERE key LIKE ?
      - close(): db.close()
    - Note: better-sqlite3 is synchronous but we wrap in async interface for future compatibility

    **file-store.ts:**
    - FileStore class:
      - Constructor validates sunDir exists
      - resolvePath(relativePath): resolve and verify it stays within sunDir (prevent traversal)
      - read: fs.readFile with catch for ENOENT
      - write: fs.mkdir for parent dir + fs.writeFile
      - exists: fs.access
      - list: fs.readdir

    **api.ts:**
    - createStateEngine(sunDir) factory function
    - Initializes directory, creates database at sunDir/state.db
    - Returns { db: StateDatabase, files: FileStore, initialize, close }
  </implementation>
