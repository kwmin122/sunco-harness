export type AgentRole = 'research' | 'planning' | 'execution' | 'verification';
export type AgentFamily = 'claude' | 'openai' | 'google' | 'custom';
export type AgentTransport = 'cli' | 'sdk';

export interface PermissionSet {
  role: AgentRole; readPaths: string[]; writePaths: string[];
  allowTests: boolean; allowNetwork: boolean; allowGitWrite: boolean; allowCommands: string[];
}

export interface AgentResult {
  providerId: string; success: boolean; outputText: string;
  artifacts: Artifact[]; warnings: string[]; usage: AgentUsage; raw?: unknown;
}

export interface AgentRequest {
  role: AgentRole; prompt: string; permissions: PermissionSet;
  expectedSchema?: unknown; timeout?: number; signal?: AbortSignal;
}

export interface AgentProvider {
  id: string; family: AgentFamily; transport: AgentTransport;
  isAvailable(): Promise<boolean>;
  execute(request: AgentRequest, context: AgentExecutionContext): Promise<AgentResult>;
}

export interface AgentRouterApi {
  run(request: AgentRequest): Promise<AgentResult>;
  crossVerify(request: AgentRequest, providerIds?: string[]): Promise<AgentResult[]>;
  getUsageSummary(): AgentUsage;
}

  <behavior>
    **permission.ts:**
    - ROLE_PERMISSIONS maps 'research' -> { readPaths: ['**'], writePaths: [], allowTests: false, allowNetwork: false, allowGitWrite: false, allowCommands: [] }
    - ROLE_PERMISSIONS maps 'planning' -> { readPaths: ['**'], writePaths: ['.planning/**', '.sun/planning/**'], allowTests: false, allowNetwork: false, allowGitWrite: false, allowCommands: [] }
    - ROLE_PERMISSIONS maps 'execution' -> { readPaths: ['**'], writePaths: ['src/**', 'packages/**', 'tests/**'], allowTests: true, allowNetwork: false, allowGitWrite: true, allowCommands: ['npm test', 'npm run build'] }
    - ROLE_PERMISSIONS maps 'verification' -> { readPaths: ['**'], writePaths: [], allowTests: true, allowNetwork: false, allowGitWrite: false, allowCommands: ['npm test'] }
    - enforcePermissions(request, role) validates request.permissions match or are stricter than ROLE_PERMISSIONS[role]
    - enforcePermissions with writePath 'src/index.ts' for research role -> throws PermissionDeniedError
    - enforcePermissions with writePath '.planning/plan.md' for planning role -> passes

    **tracker.ts:**
    - UsageTracker.record(result: AgentResult) accumulates inputTokens, outputTokens, estimatedCostUsd, wallTimeMs
    - UsageTracker.getSummary() returns totals
    - UsageTracker handles undefined token counts (CLI may not report them)

    **router.ts:**
    - createAgentRouter({ providers, context, tracker }) returns AgentRouterApi
    - run(request) selects provider by role (D-23: execution->cli, research->sdk), enforces permissions, executes, records usage
    - run(request) with no available provider -> throws ProviderUnavailableError
    - run(request) with timeout -> wraps execute in AbortSignal.timeout(), throws ExecutionTimeoutError on expiry (D-26)
    - crossVerify(request) dispatches to multiple providers via Promise.allSettled, returns all results (D-22)
    - getUsageSummary() returns accumulated usage from tracker

    **claude-cli.ts:**
    - ClaudeCliProvider.id = 'claude-code-cli', family = 'claude', transport = 'cli'
    - isAvailable(): checks `which claude` via execa, returns boolean
    - execute(request, context): spawns `claude -p --output-format json` with execa, passes prompt via stdin, parses JSON output
    - Converts claude CLI flags from PermissionSet: --allowedTools for writePaths, --model for default model
    - Handles AbortSignal: passes signal to execa for process kill
    - Normalizes output to AgentResult with estimated: true for usage (CLI token reporting is incomplete)
    - On non-zero exit: throws ProviderExecutionError

    **claude-sdk.ts:**
    - ClaudeSdkProvider.id = 'claude-sdk', family = 'claude', transport = 'sdk'
    - isAvailable(): checks if ANTHROPIC_API_KEY env var exists, returns boolean
    - execute(request, context): uses Vercel AI SDK generateText() with @ai-sdk/anthropic provider
    - Normalizes response to AgentResult with estimated: false (exact token counts)
    - Handles AbortSignal: passes to AI SDK options

    **result.ts:**
    - normalizeResult(raw, providerId): ensures consistent AgentResult shape
    - Extracts artifacts from structured output when present
  </behavior>
  <implementation>
    **permission.ts:**
    - Export const ROLE_PERMISSIONS: Record<AgentRole, Omit<PermissionSet, 'role'>> with the 4 role defaults per D-04/AGT-04
    - Export function enforcePermissions(request: AgentRequest): void -- iterate request.permissions fields, validate each against ROLE_PERMISSIONS[request.role]. Use picomatch for glob path matching. Throw PermissionDeniedError on violation.

    **tracker.ts:**
    - UsageTracker class with private accumulators
    - record(result): add result.usage values (handle undefined with ?? 0)
    - getSummary(): return { inputTokens, outputTokens, estimatedCostUsd, estimated: hasAnyEstimates, wallTimeMs }
    - persist(stateEngine): save to state engine under 'usage.session' key

    **router.ts:**
    - AgentRouter class implementing AgentRouterApi
    - Constructor receives providers Map<string, AgentProvider>, context, tracker
    - Private selectProvider(role): D-23 mapping -- execution => find cli provider, research/planning/verification => find sdk provider. Fallback to any available.
    - run: selectProvider -> enforcePermissions -> provider.execute with timeout wrapping -> tracker.record -> return result
    - Timeout wrapping: if request.timeout, create AbortSignal.timeout(request.timeout), merge with request.signal
    - crossVerify: get all providers (or specified), Promise.allSettled, collect fulfilled results, return array

    **claude-cli.ts:**
    - import { execa } from 'execa'
    - ClaudeCliProvider class
    - isAvailable: try { await execa('which', ['claude']); return true } catch { return false }
    - execute: build args array: ['-p', '--output-format', 'json', '--max-turns', '1']. Add permission flags. Spawn with execa('claude', args, { input: request.prompt, signal: request.signal ?? context.signal, timeout: request.timeout, cwd: context.cwd }). Parse JSON stdout. Normalize to AgentResult.

    **claude-sdk.ts:**
    - import { generateText } from 'ai'
    - import { anthropic } from '@ai-sdk/anthropic'
    - ClaudeSdkProvider class
    - isAvailable: return !!process.env.ANTHROPIC_API_KEY
    - execute: const { text, usage } = await generateText({ model: anthropic('claude-sonnet-4-20250514'), prompt: request.prompt, abortSignal: request.signal }). Return AgentResult with exact usage.

    **index.ts:** Re-export all public APIs
