interface LockState {
  pid: number;
  phase: number;
  step: string;
  startedAt: string;
  history: Array<{ skillId: string; success: boolean; timestamp: string }>;
}

class AutoLock {
  constructor(private sunDir: string) {}

  // Write lock file with current PID and phase/step
  async acquire(phase: number, step: string): Promise<void>

  // Update step in existing lock
  async updateStep(phase: number, step: string): Promise<void>

  // Delete lock file on clean exit
  async release(): Promise<void>

  // Check if a lock exists from a dead process
  async check(): Promise<{ exists: boolean; crashed: boolean; state: LockState | null }>

  // Append to history array in lock (for stuck detection)
  async recordInvocation(skillId: string, success: boolean): Promise<void>

  // Read current history from lock
  async getHistory(): Promise<LockState['history']>
}
```

All methods are async with graceful error handling (lock file might not exist).

}
```

Stuck detection rules:
1. **Same skill fails N times consecutively** (default N=3): stuck
2. **Oscillation**: A fails, B fails, A fails, B fails (4+ alternating failures): stuck
3. All other patterns: not stuck

No state — pure function on history array. StuckDetector is stateless.

Create `budget-guard.ts` with a BudgetGuard class:

```typescript
type BudgetStatus = 'ok' | 'warning_50' | 'warning_75' | 'warning_90' | 'exceeded';

interface BudgetCheckResult {
  status: BudgetStatus;
  currentCostUsd: number;
  ceilingUsd: number;
  percentUsed: number;
  message: string | null;
}

class BudgetGuard {
  constructor(private ceilingUsd: number | null) {}

  // Check current cost against ceiling
  check(currentCostUsd: number): BudgetCheckResult

  // Create from config (handles missing/null ceiling)
  static fromConfig(config: { budget_ceiling?: number }): BudgetGuard
}
```

Logic:
- `ceilingUsd === null` → always returns `{ status: 'ok' }` (no ceiling set)
- `currentCost >= ceiling` → `exceeded`
- `currentCost >= 0.9 * ceiling` → `warning_90`
- `currentCost >= 0.75 * ceiling` → `warning_75`
- `currentCost >= 0.5 * ceiling` → `warning_50`
- Otherwise → `ok`

Messages:
- warning_50: "Budget 50% used ($X.XX / $Y.YY)"
- warning_90: "Budget 90% used — approaching ceiling"
- exceeded: "Budget ceiling reached ($X.XX / $Y.YY) — stopping auto mode"
