  <behavior>
    roadmap-parser:
    - parseRoadmap(content: string) returns { phases: ParsedPhase[], progress: ParsedProgress[] }
    - Extracts phase list lines: "- [ ] **Phase 3: Name** - description" and "- [x] **Phase 1: Name** - description"
    - Extracts progress table rows: "| 1. Core Platform | 12/12 | Complete |"
    - Handles decimal phase numbers (3.1)
    - Returns empty arrays for malformed input (does not throw)
    - Trims captured groups

    roadmap-writer:
    - addPhase(content: string, name: string, description: string): string -- appends phase with next sequential integer number
    - insertPhase(content: string, name: string, description: string, afterPhase: number): string -- inserts as decimal (e.g., 3.1, 3.2)
    - removePhase(content: string, phaseNumber: number | string): { content: string, removed: boolean, reason?: string } -- removes if not started/completed, renumbers subsequent integers

    state-reader:
    - parseStateMd(content: string) returns ParsedState
    - Extracts YAML frontmatter between --- delimiters
    - Parses: phase, plan (from "Plan:" line in body), status (from "Status:" line in body), lastActivity, progress block

    handoff:
    - HandoffSchema is a Zod object schema
    - writeHandoff(fileStore: FileStoreApi, handoff: Handoff): writes HANDOFF.json to .sun/ root category
    - readHandoff(fileStore: FileStoreApi): reads and validates HANDOFF.json, returns null if not found

    git-state:
    - captureGitState(cwd: string): returns { branch, uncommittedChanges, uncommittedFiles }
    - Uses simple-git: simpleGit(cwd).status() for branch + file changes
    - Returns fallback { branch: 'unknown', uncommittedChanges: false, uncommittedFiles: [] } if git fails
  </behavior>
